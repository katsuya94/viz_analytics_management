def self.get_value(company)
        sources = Hash.new
        self.dependencies.each do |d|
                datum = d.get_datum(company)
                return nil if datum.nil?
                sources[d.name.underscore.to_sym] = datum.value
        end
        return self.calculate(sources)
end
